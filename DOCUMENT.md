class Car:
    """
    Класс Car представляет автомобиль с определенными характеристиками.

    Атрибуты:
        name (str): Имя владельца или название автомобиля.
        model (str): Модель автомобиля.
        age (int): Возраст автомобиля в годах.

    Методы:
        introduce_car(): Выводит информацию о машине и её владельце.
    """

    def __init__(self, name, model, age):
        """
        Инициализация нового объекта Car.

        Args:
            name (str): Имя владельца автомобиля.
            model (str): Модель автомобиля.
            age (int): Возраст автомобиля в годах.
        """
        self.name = name
        self.model = model
        self.age = age

    def introduce_car(self):
        """
        Выводит информацию о человеке и его автомобиле.
        Формат: "Человек <name>, возраст <model>, пол <age>"
        """
        print(f"Человек {self.name}, возраст {self.model}, пол {self.age}")

Что делает этот класс?

Этот класс Car моделирует автомобиль с базовыми характеристиками — имя владельца, модель и возраст. В нем есть метод introduce_car, который выводит информацию о владельце и его автомобиле в понятном формате.
